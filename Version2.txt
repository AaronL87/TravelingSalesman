  Dict = {}
  for (i,j) in CrntBndPts:
    List = []
    Crnt_Min = float()
    for (x,y) in In_Points:
      Check_Dist = sqrt((x-i)**2 + (y-j)**2)      
      if len(List) == 0:
        Crnt_Min = Check_Dist
        List.append((x,y))  
      else:
        if Crnt_Min == Check_Dist:
          List.append((x,y))          
        elif Crnt_Min >= Check_Dist:
          del List[:]
          List.append((x,y))
          Crnt_Min = Check_Dist 
        else:
          pass
    Dict.update({((i,j),Crnt_Min):List}) 
  print("Dist_Dict:",Dict,"\n")  
  return Dict
